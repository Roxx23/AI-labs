:- initialization(main).

main :-
    write('Enter vehicle name: '), nl,
    read(X),

    (vehicle(X) ->
        write('It is a vehicle'), nl
    ;
        write('Not a vehicle'), nl
    ),

    (two_wheeler(X) ->
        write('It is a two wheeler'), nl
    ;
        true
    ),

    (four_wheeler(X) ->
        write('It is a four wheeler'), nl
    ;
        true
    ),

    (heavy_vehicle(X) ->
        write('It is a heavy vehicle'), nl
    ;
        true
    ),

    (light_vehicle(X) ->
        write('It is a light vehicle'), nl
    ;
        true
    ),

    (public_transport(X) ->
        write('It is public transport'), nl
    ;
        true
    ),

    (private_vehicle(X) ->
        write('It is a private vehicle'), nl
    ;
